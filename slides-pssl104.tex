\section{Sheaf cohomology}

\begin{frame}
  \begin{columns}
    \begin{column}{0.5\textwidth}
      \centering{\usebeamercolor[fg]{item}{\textbf{\Large Sheaf cohomology}}}\bigskip

      \justifying
      Let~$E$ be a sheaf of modules over a space~$X$.
      Choose an \bad{injective resolution} $0 \to E \to I^0 \to I^1 \to \cdots$.
      Then $H^n(X; E)$ is the $n$-th cohomology of
      \[ (0 \xrightarrow{\phantom{d^0}} I^0(X) \xrightarrow{d^0} I^1(X) \xrightarrow{d^1} \cdots), \]
      that is~$\operatorname{ker}(d^n)\mathop{/}\operatorname{im}(d^{n-1})$.
    %  \begin{align*}
    %    H^n(X; E) &\defeq \text{$n$-th cohomology of $(0 \to I^0(X) \to I^1(X) \to \cdots)$} \\
    %    &\phantom{\vcentcolon}= \operatorname{ker}(I^n(X) \to I^{n+1}(X))\mathop{/}\operatorname{im}(I^{n-1}(X) \to I^n(X))
    %  \end{align*}
    %  is well-defined up to isomorphism.

      \begin{itemize}
        \item The modules~$H^n(X; E)$ are important invariants.
        \item Let~$A$ be a ring. Then $H^n(X; \underline{A}) = H_{\text{sing}}^n(X;
        A)$, if~$X$ is locally semiconnected [XXX].
      \end{itemize}
